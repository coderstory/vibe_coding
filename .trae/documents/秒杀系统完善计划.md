```java
/**
 * 分布式锁服务接口
 * 
 * 提供基于 Redisson 的分布式锁操作能力
 * 
 * @author system
 */
public interface DistributedLockService {
    
    /**
     * 获取分布式锁
     * 
     * @param lockKey 锁的键
     * @return 锁对象，获取失败返回 null
     */
    RLock getLock(String lockKey);
    
    /**
     * 尝试获取锁（带等待时间）
     * 
     * @param lockKey 锁的键
     * @param waitTime 等待时间
     * @param unit 时间单位
     * @return 是否获取成功
     */
    boolean tryLock(String lockKey, long waitTime, TimeUnit unit);
    
    /**
     * 释放锁
     * 
     * @param lockKey 锁的键
     */
    void unlock(String lockKey);
    
    /**
     * 执行带锁的业务逻辑
     * 
     * @param lockKey 锁的键
     * @param supplier 业务逻辑
     * @return 业务执行结果
     */
    <T> T executeWithLock(String lockKey, Supplier<T> supplier);
}
```

### 任务 6：完善 SSE 实时通知

#### 6.1 创建 SSE 服务
```java
/**
 * SSE 实时通知服务
 * 
 * 功能：管理 SSE 连接，提供实时消息推送
 * 
 * 核心功能：
 * 1. 订阅接口 - 建立 SSE 连接
 * 2. 推送消息 - 向指定连接推送消息
 * 3. 取消订阅 - 关闭 SSE 连接
 */
@Service
public class SseService {
    
    /** 连接池：queueId -> SSE 连接 */
    private final Map<String, SseEmitter> emitters = new ConcurrentHashMap<>();
    
    /**
     * 建立 SSE 订阅
     * 
     * @param queueId 队列ID
     * @param timeout 超时时间（毫秒）
     * @return SseEmitter
     */
    public SseEmitter subscribe(String queueId, long timeout);
    
    /**
     * 推送消息到指定队列
     * 
     * @param queueId 队列ID
     * @param data 消息数据
     */
    public void sendToQueue(String queueId, Object data);
    
    /**
     * 取消订阅
     * 
     * @param queueId 队列ID
     */
    public void unsubscribe(String queueId);
}
```

#### 6.2 创建 SSE Controller
```java
/**
 * 秒杀 SSE 订阅控制器
 * 
 * 提供 SSE 实时通知接口
 */
@RestController
@RequestMapping("/api/seckill")
public class SeckillSseController {
    
    /**
     * 订阅秒杀结果通知
     * 
     * @param queueId 队列ID
     * @return SSE 事件流
     */
    @GetMapping(value = "/subscribe/{queueId}", produces = MediaType.TEXT_EVENT_STREAM_VALUE)
    public SseEmitter subscribe(@PathVariable String queueId);
}
```

#### 6.3 集成到秒杀流程
1. 抢购请求 → 返回 queueId
2. 前端建立 SSE 连接
3. 后端异步处理
4. 处理完成 → SSE 推送结果
5. 前端收到结果 → 展示

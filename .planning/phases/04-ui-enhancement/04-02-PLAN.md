```vue
<script setup>
import { useRouter } from 'vue-router'
import { useUserStore } from '@/store/user'
import { useThemeStore } from '@/store/theme'

const router = useRouter()
const userStore = useUserStore()
const themeStore = useThemeStore()

async function handleLogout() {
  // 使用 Vuestic toast 或 modal 替代
  try {
    const confirmed = await this.$va.modal.confirm({
      message: '确定要退出登录吗？',
      okText: '确定',
      cancelText: '取消'
    })
    if (confirmed) {
      await userStore.logout()
      router.push('/login')
    }
  } catch (e) {
    // 用户取消
  }
}

function toggleTheme() {
  themeStore.toggleTheme()
  themeStore.syncTheme()
}
</script>

<template>
  <div class="app-header">
    <div class="header-right">
      <va-button flat @click="toggleTheme" class="theme-btn">
        {{ themeStore.isDark ? '☀️' : '🌙' }}
      </va-button>
      <va-dropdown>
        <va-button preset="secondary">
          <va-avatar size="small" /> {{ userStore.userInfo?.name || userStore.userInfo?.username }}
        </va-button>
        <va-dropdown-menu>
          <va-dropdown-item @click="handleLogout">退出登录</va-dropdown-item>
        </va-dropdown-menu>
      </va-dropdown>
    </div>
  </div>
</template>
```

  <files>app-vue/src/views/auth/Login.vue</files>
  <read_first>app-vue/src/views/auth/Login.vue</read_first>
  <action>
重构 Login.vue，将 Element Plus 替换为 Vuestic UI:

替换映射：
- `el-form` → HTML form + va-input
- `el-form-item` → div wrapper
- `el-input` → `va-input`
- `el-button` → `va-button`
- `ElMessage.error` → `this.$vaToast`

关键变更：
```vue
<script setup>
import { ref, reactive } from 'vue'
import { useRouter } from 'vue-router'
import { useUserStore } from '@/store/user'
import { useThemeStore } from '@/store/theme'

const router = useRouter()
const userStore = useUserStore()
const themeStore = useThemeStore()

const loginForm = reactive({
  username: '',
  password: ''
})

const loading = ref(false)

async function handleLogin() {
  if (!loginForm.username || !loginForm.password) {
    this.$vaToast.warning('请输入用户名和密码')
    return
  }

  loading.value = true
  try {
    await userStore.login(loginForm.username, loginForm.password)
    themeStore.loadTheme()
    router.push('/dashboard')
  } catch (error) {
    this.$vaToast.error(error.message || '用户名或密码错误')
  } finally {
    loading.value = false
  }
}

function handleKeydown(e) {
  if (e.key === 'Enter') {
    handleLogin()
  }
}
</script>

<template>
  <div class="login-container dynamic-gradient" @keydown="handleKeydown">
    <div class="login-particles">
      <div class="particle floating" style="top: 10%; left: 20%;"></div>
      <div class="particle floating" style="top: 60%; left: 80%; animation-delay: 0.5s;"></div>
      <div class="particle floating" style="top: 80%; left: 10%; animation-delay: 1s;"></div>
    </div>

    <div class="login-card glass glow">
      <div class="login-header">
        <h1 class="login-title">管理系统</h1>
        <p class="login-subtitle">欢迎回来</p>
      </div>

      <form @submit.prevent="handleLogin" class="login-form">
        <va-input
          v-model="loginForm.username"
          placeholder="请输入用户名"
          class="glass-input"
        />

        <va-input
          v-model="loginForm.password"
          type="password"
          placeholder="请输入密码"
          class="glass-input"
        />

        <va-button
          type="submit"
          :loading="loading"
          class="login-button glass-button"
          @click="handleLogin"
        >
          登 录
        </va-button>
      </form>

      <div class="login-footer">
        <span class="theme-toggle" @click="themeStore.toggleTheme">
          {{ themeStore.isDark ? '☀️ 亮色模式' : '🌙 暗色模式' }}
        </span>
      </div>
    </div>
  </div>
</template>
```

public List<Map<String, Object>> getMessageList(String topic, long startTime, long endTime, int maxMsg) {
    String brokerAddr = getFirstBrokerAddr(topic);
    if (brokerAddr == null) {
        throw BusinessException.badRequest("未找到可用的 Broker");
    }

    try {
        QueryResult queryResult = defaultMQAdminExt.queryMessage(
            topic,
            "*", // msgId 模糊匹配，* 表示所有
            maxMsg > 0 ? maxMsg : 100,
            startTime,
            endTime
        );

        List<Map<String, Object>> result = new ArrayList<>();
        for (MessageExt msg : queryResult.getMessageList()) {
            Map<String, Object> item = new HashMap<>();
            item.put("msgId", msg.getMsgId());
            item.put("topic", msg.getTopic());
            item.put("tags", msg.getTags());
            item.put("keys", msg.getKeys());
            item.put("timestamp", msg.getStoreTimestamp());
            item.put("queueId", msg.getQueueId());
            item.put("queueOffset", msg.getQueueOffset());
            item.put("properties", msg.getProperties());
            result.add(item);
        }
        return result;
    } catch (Exception e) {
        log.error("查询消息列表失败: topic={}", topic, e);
        throw BusinessException.badRequest("查询消息失败: " + e.getMessage());
    }
}

public Map<String, Object> getMessageDetail(String topic, String msgId) {
    String brokerAddr = getFirstBrokerAddr(topic);
    if (brokerAddr == null) {
        throw BusinessException.badRequest("未找到可用的 Broker");
    }

    try {
        // 通过 queryMessage 查询单条消息
        QueryResult queryResult = defaultMQAdminExt.queryMessage(
            topic,
            msgId,
            1, // maxMsg=1 只返回一条
            System.currentTimeMillis() - 7 * 24 * 60 * 60 * 1000, // 7天前
            System.currentTimeMillis()
        );

        for (MessageExt msg : queryResult.getMessageList()) {
            if (msg.getMsgId().equals(msgId)) {
                Map<String, Object> result = new HashMap<>();
                result.put("msgId", msg.getMsgId());
                result.put("topic", msg.getTopic());
                result.put("tags", msg.getTags());
                result.put("keys", msg.getKeys());
                result.put("timestamp", msg.getStoreTimestamp());
                result.put("queueId", msg.getQueueId());
                result.put("queueOffset", msg.getQueueOffset());
                result.put("properties", msg.getProperties());
                result.put("body", new String(msg.getBody(), StandardCharsets.UTF_8));
                return result;
            }
        }
        throw BusinessException.notFound("未找到消息: " + msgId);
    } catch (Exception e) {
        log.error("查询消息详情失败: topic={}, msgId={}", topic, msgId, e);
        throw BusinessException.badRequest("查询消息详情失败: " + e.getMessage());
    }
}

@Override
public Map<String, Object> getMessageTrace(String topic, String msgId) {
    // RocketMQ 5.x 轨迹数据存储在 RMQ_SYS_TRACE_TOPIC
    // 使用 searchOffset 获取消息的物理位点，然后组合消费进度信息
    try {
        Map<String, Object> result = new HashMap<>();
        result.put("msgId", msgId);
        result.put("topic", topic);

        // 查询消费轨迹（简化实现，实际需要查询 RMQ_SYS_TRACE_TOPIC）
        List<Map<String, Object>> consumeTraceList = new ArrayList<>();

        // 获取该 topic 下的所有 Consumer Group
        List<String> groups = getConsumerGroupList();
        for (String group : groups) {
            try {
                // 检查该 group 是否消费过此消息
                ConsumeStats consumeStats = defaultMQAdminExt.getConsumeStats(group, topic);
                if (consumeStats != null && consumeStats.getOffsetTable() != null) {
                    // 简化：实际轨迹需要查询 trace topic
                    // 这里返回一个占位结构
                    Map<String, Object> trace = new HashMap<>();
                    trace.put("consumerGroup", group);
                    trace.put("status", "已消费");
                    trace.put("consumeTime", System.currentTimeMillis());
                    consumeTraceList.add(trace);
                }
            } catch (Exception e) {
                // 跳过无法获取的 group
            }
        }
        result.put("consumeTraceList", consumeTraceList);
        return result;
    } catch (Exception e) {
        log.error("查询消息轨迹失败: topic={}, msgId={}", topic, msgId, e);
        throw BusinessException.badRequest("查询消息轨迹失败: " + e.getMessage());
    }
}

/**
 * 消息列表项

export interface MessageVO {
  msgId: string
  topic: string
  tags: string
  keys: string
  timestamp: number
  queueId: number
  queueOffset: number
  properties?: Record<string, string>
}

/**
 * 消息详情
 */
export interface MessageDetailVO extends MessageVO {
  body: string
}

/**
 * 消息轨迹
 */
export interface MessageTraceVO {
  msgId: string
  topic: string
  consumeTraceList: {
    consumerGroup: string
    status: string
    consumeTime: number
  }[]
}

/**
 * 查询消息列表
 */
export function getMessageList(topic: string, startTime: number, endTime: number, maxMsg = 100) {
  return request.get<MessageVO[]>('/rocketmq/messages', {
    params: { topic, startTime, endTime, maxMsg }
  })
}

/**
 * 查询消息详情
 */
export function getMessageDetail(topic: string, msgId: string) {
  return request.get<MessageDetailVO>(`/rocketmq/messages/${encodeURIComponent(topic)}/${msgId}`)
}

/**
 * 查询消息轨迹
 */
export function getMessageTrace(topic: string, msgId: string) {
  return request.get<MessageTraceVO>(`/rocketmq/messages/${encodeURIComponent(topic)}/${msgId}/trace`)
}

<script setup lang="ts">
import { ref, watch } from 'vue'
import { getMessageDetail, type MessageDetailVO } from '@/api/rocketmq'
import { ElMessage } from 'element-plus'

const props = defineProps<{
  modelValue: boolean
  message: MessageDetailVO | null
}>()

const emit = defineEmits<{
  'update:modelValue': [value: boolean]
}>()

const loading = ref(false)
const detail = ref<MessageDetailVO | null>(null)
const bodyExpanded = ref(false)
const MAX_BODY_LENGTH = 200

watch(() => props.modelValue, async (val) => {
  if (val && props.message) {
    loading.value = true
    try {
      const res = await getMessageDetail(props.message.topic, props.message.msgId)
      detail.value = res.data
    } catch (e: any) {

      loading.value = false
    }
  }

const close = () => {
  emit('update:modelValue', false)
}

const formatTime = (timestamp: number) => {
  return new Date(timestamp).toLocaleString('zh-CN')
}

const getBodyDisplay = () => {
  if (!detail.value?.body) return ''
  if (bodyExpanded.value || detail.value.body.length <= MAX_BODY_LENGTH) {
    return detail.value.body
  }
  return detail.value.body.substring(0, MAX_BODY_LENGTH) + '...'
}

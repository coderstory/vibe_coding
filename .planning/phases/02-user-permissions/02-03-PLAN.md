  <action>
    创建 app-vue/src/api/role.js：

    ```javascript
    import request from '@/utils/request'

    export function getRoleList(params) {
      return request({
        url: '/api/roles',
        method: 'get',
        params
      })
    }

    export function getRoleDetail(id) {
      return request({
        url: `/api/roles/${id}`,
        method: 'get'
      })
    }

    export function createRole(data) {
      return request({
        url: '/api/roles',
        method: 'post',
        data
      })
    }

    export function updateRole(id, data) {
      return request({
        url: `/api/roles/${id}`,
        method: 'put',
        data
      })
    }

    export function deleteRole(id) {
      return request({
        url: `/api/roles/${id}`,
        method: 'delete'
      })
    }

    export function getRoleMenus(roleId) {
      return request({
        url: `/api/roles/${roleId}/menus`,
        method: 'get'
      })
    }

    export function assignRoleMenus(roleId, menuIds) {
      return request({
        url: `/api/roles/${roleId}/menus`,
        method: 'put',
        data: menuIds
      })
    }

    export function getMenuTree() {
      return request({
        url: '/api/menus/tree',
        method: 'get'
      })
    }
    ```

<task type="auto">
  <name>task 3: 改造AppMenu.vue实现动态菜单权限过滤 (MENU-02)</name>
  <files>app-vue/src/components/AppMenu.vue</files>
  <read_first>app-vue/src/components/AppMenu.vue</read_first>
  <action>
    改造AppMenu.vue实现用户只能看到有权限访问的菜单项（MENU-02）。

    当前实现：AppMenu渲染所有菜单
    目标实现：根据当前登录用户的角色权限，动态过滤显示菜单

    实现方案：

    1. 在AppMenu组件中：
       - 登录成功后，前端已存储用户信息（包含userId和roleId）
       - 调用 GET /api/menus/user/{userId} 获取用户可见菜单
       - 将返回的菜单数据存储在响应式变量中
       - 菜单渲染使用过滤后的数据

    2. 菜单过滤逻辑：
       - 如果用户是超级管理员（roleId=1），返回所有菜单
       - 普通用户根据角色-菜单权限返回对应的菜单树

    3. 菜单数据更新时机：
       - 用户登录后首次加载
       - 用户被分配不同角色后（可调用刷新接口）

    4. 前端代码结构参考：

    ```javascript
    <script setup>
    import { ref, onMounted } from 'vue'
    import { useUserStore } from '@/stores/user' // 如果有userStore
    import { getUserMenus } from '@/api/menu'    // 需要创建menu API

    const visibleMenus = ref([])

    onMounted(async () => {
      const userInfo = JSON.parse(localStorage.getItem('userInfo'))
      if (userInfo && userInfo.id) {
        const res = await getUserMenus(userInfo.id)
        visibleMenus.value = res.data || []
      }
    })
    </script>

    <template>
      <!-- 使用 visibleMenus 替代原来的完整菜单数据 -->
      <el-menu>
        <template v-for="menu in visibleMenus" :key="menu.id">
          <!-- 递归渲染菜单 -->
        </template>
      </el-menu>
    </template>
    ```

    注意：如果没有menu.js API文件，需要创建 app-vue/src/api/menu.js：

    ```javascript
    import request from '@/utils/request'

    export function getUserMenus(userId) {
      return request({
        url: `/api/menus/user/${userId}`,
        method: 'get'
      })
    }

    export function getMenuTree() {
      return request({
        url: '/api/menus/tree',
        method: 'get'
      })
    }
    ```
  </action>
